var _            = require('underscore'),
    EventEmitter = require('events').EventEmitter;

/**
 * @constructor
 * @augments EventEmitter
 * @param {string} [namespace=''] Namespace
 */
function DataMap(namespace, locals) {
	this._namespace = namespace || '';
	this._locals = locals || [];
}

DataMap.prototype = new EventEmitter();
DataMap.prototype.constructor = DataMap;

/**
 * Determine if we're storing the values within this object, or the client
 * @return {boolean} namespaced
 */
DataMap.prototype._isNamespaced = function() {
	return _.isString(this._namespace) && _.isObject(this.client) && this.client !== this;
};



/**
 * Set the value of an option
 * @param {string} option
 * @param {mixed} value
 * @return this
 */
DataMap.prototype.set = function(option, value) {
	if (this._isNamespaced())
		this.client.set(this._namespace + '.' + option, value);
	else if (this.has(option) && !_.isUndefined(value))
		this._set[option] = value;

	return this;
};

/**
 * Get the value of an option
 * @param {string} option
 * @return {mixed|undefined} value Undefined if option does not exist
 */
DataMap.prototype.get = function(option) {
	if (!this.has(option))
		return undefined;

	return this._set[option];
};

/**
 * Find out if an option exists
 * @param {string} option
 * @return {boolean} exists
 */
DataMap.prototype.has = function(option) {
	if (!_.isObject(this._set))
		this._set = {};

	return !_.isUndefined(this._set[option]);
};

exports.DataMap = DataMap;
